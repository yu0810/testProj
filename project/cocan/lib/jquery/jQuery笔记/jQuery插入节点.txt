append()		//向每个匹配的元素内部追加内容

appendTo()	//将所有匹配的元素追加到指定的元素中
		//这种方法颠倒了常规的$(A).append(B)的操作
		//即不是把B追加到A中，而是把A追加到B中。

prepend()		//向每个匹配的元素内部前置内容

prependTo()	//将所有匹配的元素前置到指定的元素中
		//这种方法颠倒了常规的$(A).prepend(B)的操作
		//即不是把B前置到A中，而是把A前置到B中。

after()		//在每个匹配的元素之后插入内容

insertAfter()	//将所有匹配的元素插入到指定元素的后面
		//这种方法颠倒了常规的$(A).after(B)的操作
		//即不是把B插入到A后面，而是把A插入到B后面。

before()		//在每个匹配的元素之前插入内容		

insertBefore()	//将所有匹配的元素插入到指定元素的前面
		//这种方法颠倒了常规的$(A).bofore(B)的操作
		//即不是把B插入到A前面，而是把A插入到B前面。
通俗点讲 append和prepend是发生在父子关系中	after和before发生在兄弟关系中
